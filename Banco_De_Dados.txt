create database npxconsole;
use npxconsole;

create table pessoa(
	pessoa_id int auto_increment primary key,
    email varchar(40) not null,
    senha varchar(50) not null
);

create table produto(
	produto_id int auto_increment primary key,
    marca varchar(10),
    nome varchar(30),
    preco double,
    descricao varchar(500),
    principal boolean,
    img varchar(100),
    dia datetime
);
